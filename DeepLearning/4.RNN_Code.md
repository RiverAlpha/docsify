# RNN代码

```python

import torch
import torch.nn as nn

from torch.autograd import forward_ad
class MyRNN(nn.Module):
  def __init__(self, input_size, hidden_size, output_size):
    """
      参数:
      input_size: 每个时间步输入特征的维度
      hidden_size: 隐藏状态的维度 (RNN的记忆容量)
      output_size: 每个时间步输出特征的维度
    """
    # super(MyRNN,self).__init__()
    super().__init__()
    self.hidden_size = hidden_size

    ## 初始化参数
    self.W_xh = nn.Parameter(torch.randn(input_size,hidden_size) * 0.01)
    self.W_hh = nn.Parameter(torch.randn(hidden_size,hidden_size) * 0.01)
    self.W_hq = nn.Parameter(torch.randn(hidden_size,output_size) * 0.01)

    ## 初始化状态的偏置
    self.b_h = nn.Parameter(torch.zeros(hidden_size))
    self.b_q = nn.Parameter(torch.zeros(output_size))


  def forward(self,X,state=None):
    """
    X 的形状通常是: (batch_size, seq_len, input_size)
    state 是初始的隐藏状态
    """

    batch_size, seq_len, _ = X.size()

    # 初始情况没有初始状态Ht-1
    if state is None:
      state = torch.zeros(batch_size,self.hidden_size,device=X.device)

    outputs = []
    # 沿着序列长度 (seq_len) 进行循环，模拟时间推移
    for t in range(seq_len):
      # 取出当前时间步 t 的输入，形状变为 (batch_size, input_size)
      # 取出batch个t位置的向量
      X_t = X[:,t,:]

      # 计算新的Ht
      state = torch.tanh(torch.matmul(X_t, self.W_xh) + torch.matmul(state, self.W_hh) + self.b_h)

      # 计算输出
      out_t = torch.matmul(state, self.W_hq) + self.b_q

      # 保存结果
      outputs.append(out_t)
    
    return torch.stack(outputs,dim=1), state
```

## 验证

```python
# 设定超参数
batch_size = 32      # 批量大小
seq_len = 10         # 序列长度 (例如一句话有10个单词)
input_size = 20      # 输入维度 (例如每个单词用长20的词向量表示)
hidden_size = 64     # 隐藏层维度
output_size = 5      # 输出维度 (例如做5分类任务)

# 1. 实例化模型
model = MyRNN(input_size, hidden_size, output_size)

# 2. 生成随机的输入数据
# 形状必须符合 forward 方法的要求: (batch_size, seq_len, input_size)
X = torch.randn(batch_size, seq_len, input_size)

# 3. 前向传播
outputs, final_state = model(X)

print(f"输入数据形状: {X.shape}")
print(f"全部时间步的输出形状: {outputs.shape}")
print(f"最后一个时间步的隐藏状态形状: {final_state.shape}")
```

## 优化器

```python
import torch.optim as optim

# 1. 生成假的标签数据 Y
# 形状: (batch_size, seq_len) -> 每个时间步对应一个正确的类别索引 (0 到 output_size-1)
# 承接上文：batch_size=32, seq_len=10, output_size=5
Y = torch.randint(0, output_size, (batch_size, seq_len))

# 2. 定义损失函数
# 对于多分类任务，标准选择是交叉熵损失 (Cross Entropy Loss)
criterion = nn.CrossEntropyLoss()

# 3. 定义优化器
# 我们使用最常用的 Adam 优化器，并将模型中所有可学习的参数 (W_xh, W_hh, b_h 等) 传给它
# lr=0.01 是学习率，决定了每次参数更新的步子迈多大
optimizer = optim.Adam(model.parameters(), lr=0.01)
```

## 训练

```python
epochs = 100  # 训练的总轮数

for epoch in range(epochs):
    # 【第1步：清空梯度】
    # PyTorch 默认会累加梯度。每次开始新一轮计算前，必须把上一次的梯度清零，否则会算错。
    optimizer.zero_grad()
    
    # 【第2步：前向传播 (Forward)】
    # 让输入 X 经过我们手写的 RNN，得到所有时间步的预测输出
    # outputs 的形状: (batch_size, seq_len, output_size)
    outputs, _ = model(X)
    
    # 【第3步：计算损失 (Compute Loss)】
    # ⚠️ 关键坑点：
    # nn.CrossEntropyLoss 默认期望的输入形状是二维的 (N, C)，其中 N 是样本总数，C 是类别数。
    # 期望的标签 Y 是以为一维的 (N)。
    # 但我们的 outputs 是三维的！所以我们需要把 "批次" 和 "序列长度" 这两个维度合并展开。
    
    # view(-1, ...) 中 -1 的意思是：让 PyTorch 自己算出这维应该是多少
    outputs_flat = outputs.view(-1, output_size)  # 形状变为: (32 * 10, 5) -> (320, 5)
    Y_flat = Y.view(-1)                           # 形状变为: (32 * 10) -> (320)
    
    # 计算预测值和真实值之间的差异
    loss = criterion(outputs_flat, Y_flat)
    
    # 【第4步：反向传播 (Backward)】
    # PyTorch 的自动求导机制会顺着计算图，自动计算出 loss 对模型每一个权重的梯度 (导数)
    loss.backward()
    
    # 【第5步：更新权重 (Step)】
    # 优化器根据刚才算出的梯度，悄悄地修改了 W_xh, W_hh 等参数的值，让 loss 下次能小一点
    optimizer.step()
    
    # 每隔 10 轮打印一次当前的 Loss，观察模型是否在学习
    if (epoch + 1) % 10 == 0:
        print(f'Epoch [{epoch+1}/{epochs}], Loss: {loss.item():.4f}')
```
